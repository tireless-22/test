#include<stdio.h>
#include<math.h>
float Yr[8],Yi[8],Xr[8],Xi[8],ar[8],ai[8];
float br[8],bi[8],wr[4],wi[4],p[4],q[4],xi[8],xr[8];
int k;
void main()
{  printf(" Enter input sequence\n");
for(k=0;k<8;k++)
{  scanf("%f%f",&Yr[k],&Yi[k]);
  Xr[k]=Yr[k];
  Xi[k]=-Yi[k];    }
for(k=0;k<2;k++)
{  ar[k*4]=Xr[k]+Xr[k+4];
   ai[k*4]=Xi[k]+Xi[k+4];
   ar[k*4+1]=Xr[k]-Xr[k+4];
   ai[k*4+1]=Xi[k]-Xi[k+4];
   ar[k*4+2]=Xr[k+2]+Xr[k+6];
   ai[k*4+2]=Xi[k+2]+Xi[k+6];
   ar[k*4+3]=Xr[k+2]-Xr[k+6];
   ai[k*4+3]=Xi[k+2]-Xi[k+6]; }
for(k=0;k<8;k=k+4)
{  br[k]=ar[k]+ar[k+2];
   bi[k]=ai[k]+ai[k+2];
   br[k+1]=ar[k+1]+ai[k+3];
   bi[k+1]=ai[k+1]-ar[k+3];
   br[k+2]=ar[k]-ar[k+2];
   bi[k+2]=ai[k]-ai[k+2];
   br[k+3]=ar[k+1]-ai[k+3];
   bi[k+3]=ai[k+1]+ar[k+3];  }
for(k=0;k<4;k++)
{  wr[k]=cos(0.785*k);
   wi[k]=-sin(0.785*k);
   p[k]=br[k+4]*wr[k]-bi[k+4]*wi[k];
   q[k]=bi[k+4]*wr[k]+br[k+4]*wi[k];
   xr[k]=br[k]+p[k];
   xi[k]=bi[k]+q[k];
   xr[k+4]=br[k]-p[k];
   xi[k+4]=bi[k]-q[k];  }
printf("\n 8point IFFT");
for(k=0;k<8;k++)
{  xr[k]=xr[k]/8;
   xi[k]=-xi[k]/8;
printf("\n %.3f+(%.3f)",xr[k],xi[k]); 
 }}
