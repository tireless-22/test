#include<stdio.h>
int x1[15],X2[15],h[15],y[15];
int m,n,i,j;
void main()
{
printf("Enter sizes of  2 sequences m,n:\n");
scanf("%d%d",&m,&n);
printf("Enter first sequence x1:\n");
for(i=0;i<m;i++)
scanf("%d",&x1[i]);
printf("Enter second sequence x2:\n");
for(i=0;i<n;i++)
scanf("%d",&x2[i]);
j=0;
for(i=n-1;i>=0;i--)
{
  if(j<n)
  h[j]=x2[i];
  j++;
}
for(i=m;i<m+n-1;i++)
x1[i]=0;
for(i=n;i<m+n-1;i++)
h[i]=0;
for(i=0;i<m+n-1;i++)
{
y[i]=0;
for(j=0;j<=i;j++)
y[i]=(x1[j]*h[i-j])+y[i];
printf("%d\t",y[i]);
}
}
